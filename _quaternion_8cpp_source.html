192 void CQuaternion::setFromEuler(float fAngleRadiansX, float fAngleRadiansY, float fAngleRadiansZ)
193 {
194 double dSY = sin(fAngleRadiansZ * 0.5f);
195 double dSP = sin(fAngleRadiansY * 0.5f);
196 double dSR = sin(fAngleRadiansX * 0.5f);
197 double dCY = cos(fAngleRadiansZ * 0.5f);
198 double dCP = cos(fAngleRadiansY * 0.5f);
199 double dCR = cos(fAngleRadiansX * 0.5f);
200
201 q[0] = float(dSR * dCP * dCY - dCR * dSP * dSY);
202 q[1] = float(dCR * dSP * dCY + dSR * dCP * dSY);
203 q[2] = float(dCR * dCP * dSY - dSR * dSP * dCY);
204 q[3] = float(dCR * dCP * dCY + dSR * dSP * dSY);
205 normalise();
206 }

208 CVector3f CQuaternion::getEuler(void) const
209 {
210 float fSq[4];
211 fSq[0] = q[0] * q[0];
212 fSq[1] = q[1] * q[1];
213 fSq[2] = q[2] * q[2];
214 fSq[3] = q[3] * q[3];
215
216 double m11 = fSq[3] + fSq[0] - fSq[1] - fSq[2];
217 double m21 = 2 * (q[0] * q[1] + q[2] * q[3]);
218 double m31 = 2 * (q[2] * q[0] - q[1] * q[3]);
219 double m32 = 2 * (q[2] * q[1] + q[0] * q[3]);
220 double m33 = fSq[3] - fSq[0] - fSq[1] + fSq[2];
221 return CVector3f(float(atan2(m32, m33)), asinf(float(-m31)), float(atan2(m21, m11)));
222 }

224 void CQuaternion::setFromMatrix(const CMatrix& m)
225 {
226 float fTr = m.m[0] + m.m[5] + m.m[9] + 1.0f;
227 // If the trace of the matrix is greater than 0, perform an instant calculation
228 if (fTr > 0)
229 {
230 float fS = 0.5f / sqrtf(fTr);
231
232 q[3] = 0.25f / fS;
233 q[0] = (m.m[9] - m.m[6]) * fS;
234 q[1] = (m.m[2] - m.m[8]) * fS;
235 q[2] = (m.m[4] - m.m[1]) * fS;
236 }
237 // if the first diagonal element is the largest
238 else if (m.m[0] > m.m[5] && m.m[0] > m.m[10])
239 {
240 float fS = float(1 / (sqrt(m.m[0] - m.m[5] - m.m[10] + 1.0f) * 2));
241 q[0] = 0.5f * fS;
242 q[1] = (m.m[1] + m.m[4]) * fS;
243 q[2] = (m.m[2] + m.m[8]) * fS;
244 q[3] = (m.m[6] + m.m[9]) * fS;
245 }
246 // if the second.....
247 else if (m.m[5] > m.m[10])
248 {
249 float fS = float(1 / (sqrt(m.m[5] - m.m[0] - m.m[10] + 1.0f) * 2));
250 q[0] = (m.m[1] + m.m[4]) * fS;
251 q[1] = 0.5f * fS;
252 q[2] = (m.m[6] + m.m[9]) * fS;
253 q[3] = (m.m[2] + m.m[8]) * fS;
254 }
255 // if the last element is the largest
256 else
257 {
258 float fS = float(1 / (sqrt(m.m[10] - m.m[0] - m.m[5] + 1.0f) * 2));
259 q[0] = (m.m[2] + m.m[8]) * fS;
260 q[1] = (m.m[6] + m.m[9]) * fS;
261 q[2] = 0.5f * fS;
262 q[3] = (m.m[1] + m.m[4]) * fS;

272 CQuaternion CQuaternion::getSLERP(const CQuaternion& quat, float fInterval) const
273 {
274 CQuaternion q(quat);
275
276 float fDot = q[0] * q.q[0] + q[1] * q.q[1] + q[2] * q.q[2] + q[3] * q.q[3];
277 if (fDot < 0.0f)
278 {
279 q = q.getNegative();
280 fDot = -fDot;
281 }
282
283 // Inaccurate, use lerp instead
284 if (fDot < 1.00001f && fDot > 0.99999f)
285 {
286 return getLERP(q, fInterval);
287 }
288
289 // Calculate the angle between the quaternions
290 float fTheta = acosf(fDot);
291
292 return ((*this * sinf(fTheta * (1 - fInterval)) + q * sinf(fTheta * fInterval)) / sinf(fTheta));
293 }
