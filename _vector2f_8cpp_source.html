178 float CVector2f::getAngleDegreesSigned(const CVector2f& vOther) const
179 {
180 // Normalise this vector
181 CVector2f vThisNorm(x, y);
182 vThisNorm.normalise();
183
184 // Normalise other vector
185 CVector2f vOtherNorm(vOther.x, vOther.y);
186 vOtherNorm.normalise();
187
188 // Compute dot
189 float fTmp = vThisNorm.x * vOtherNorm.x + vThisNorm.y * vOtherNorm.y;
190
191 // Convert to unsigned radians
192 // make sure fTmp is valid
193 if (fTmp < -1.0f)
194 fTmp = -1.0f;
195 else if (fTmp > 1.0f)
196 fTmp = 1.0f;
197 fTmp = (float)acosf(fTmp);
198
199 // Convert to signed radians
200 if (vThisNorm.y * vOtherNorm.x < vThisNorm.x * vOtherNorm.y)
201 fTmp *= -1.0f;
202
203 // Convert to degrees
204 // NOTE: 180.0f / Pi is used to convert radians to degrees
205 // and it is a value of : 57.2957795
206 return fTmp * 57.2957795f;
207 }

209 float CVector2f::getAngleDegrees360(void) const
210 {
211 // Normalise this vector
212 CVector2f vThisNorm(x, y);
213 vThisNorm.normalise();
214
215 CVector2f vNorth(0.0f, 1.0f);
216
217 // Compute dot
218 float fTmp = vThisNorm.x * vNorth.x + vThisNorm.y * vNorth.y;
219
220 // Convert to unsigned radians
221 // First, make sure fTmp is valid
222 if (fTmp < -1.0f)
223 fTmp = -1.0f;
224 else if (fTmp > 1.0f)
225 fTmp = 1.0f;
226 fTmp = (float)acosf(fTmp);
227
228 // Convert to degrees
229 // NOTE: 180.0f / Pi is used to convert radians to degrees
230 // and it is a value of : 57.2957795
231 fTmp *= 57.2957795f;
232
233 if (vThisNorm.y * vNorth.x < vThisNorm.x * vNorth.y)
234 {
235 return fTmp;
236 }
237 return 180.0f + (180.0f - fTmp);
