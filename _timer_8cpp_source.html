30 void CTimer::update(void)
31 {
32 _mdTimePointNew = std::chrono::steady_clock::now();
33 _mdTimeDeltaSec = _mdTimePointNew - _mdTimePointOld;
34 _mdTimePointOld = _mdTimePointNew;// std::chrono::steady_clock::now();
35
36 _mdDeltaSec = _mdTimeDeltaSec.count();
37
38 // Compute FPS
39 ++_muiNumFrames;
40 _mdFPSFrameTime += _mdDeltaSec * 1000.0;
41 if (_mdFPSFrameTime > 0.0)
42 {
43 _mdFPS = _muiNumFrames * (1000.0 / _mdFPSFrameTime);
44
45 _muiNumFrames = 0;
46 _mdFPSFrameTime = 0.0;
47 }
48
49 // Compute FPS smoothed
50 _mdFPSAveragedTimeCount += _mdDeltaSec;
51 ++_miFPSAveragedNumCallsPerSec;
52 _mdFPSAveragedAccum += _mdFPS;
53 if (_mdFPSAveragedTimeCount > _mdFPSAveragedRate)
54 {
55 if (_miFPSAveragedNumCallsPerSec < 1)
56 _miFPSAveragedNumCallsPerSec = 1;
57 _mdFPSAveraged = _mdFPSAveragedAccum / _miFPSAveragedNumCallsPerSec;
58 _mdFPSAveragedTimeCount = 0;
59 _miFPSAveragedNumCallsPerSec = 0;
60 _mdFPSAveragedAccum = 0;
61 }
62
63 // Runtime
64 _mdRuntimeInSeconds += _mdDeltaSec;
65
66 // Minimum and maximum acheived framerate
67 _mdFPSMinMaxDelay -= _mdDeltaSec;
68 if (_mdFPSMinMaxDelay <= 0.0)
69 {
70 if (_mbFPSMinMaxInit)
71 {
72 _mbFPSMinMaxInit = false;
73 _mdFPSMinimum = 999999;
74 _mdFPSMaximum = 0;
75 }
76 if (!_mbFPSMinMaxInit)
77 {
78 _mdFPSMinMaxDelay = 0.0;
79 if (_mdFPSMinimum > _mdFPS)
80 _mdFPSMinimum = _mdFPS;
81 if (_mdFPSMaximum < _mdFPS)
82 _mdFPSMaximum = _mdFPS;
83 }
84 }

165 void CTimer::getClock(float& fSeconds, int& iMinutes, int& iHours, int& iDays, int& iWeeks) const
166 {
167 fSeconds = 0.0f;
168 iMinutes = 0;
169 iHours = 0;
170 iDays = 0;
171 iWeeks = 0;
172
173 // How many seconds are in a week?
174 // 60seconds in a minute, 60 minutes in an hour, 24 hours in a day and 7 days in a week.
175 // 60 * 60 = 3600 seconds per hour
176 // 3600 * 24 = 86400 seconds in a day
177 // 86400 * 7 = 604800 seconds in a week
178 double seconds = _mdRuntimeInSeconds;
179 while (seconds >= 604800) // Weeks
180 {
181 seconds -= 604800;
182 iWeeks++;
183 }
184 while (seconds >= 86400) // Days
185 {
186 seconds -= 86400;
187 iDays++;
188 }
189 while (seconds >= 3600) // Hours
190 {
191 seconds -= 3600;
192 iHours++;
193 }
194 while (seconds >= 60) // Minutes
195 {
196 seconds -= 60;
197 iMinutes++;
198 }
199 fSeconds = (float)seconds; // Seconds
200 }
