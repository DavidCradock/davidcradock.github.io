28 void CColouruc::setFromHSB(float fHueAmount, float fSaturation, float fBrightness)
29 {
30 // Clamp each passed value
31 clamp(fHueAmount, 0.0f, 1.0f);
32 clamp(fSaturation, 0.0f, 1.0f);
33 clamp(fBrightness, 0.0f, 1.0f);
34
35 // Compute colour
36 fHueAmount *= 360.0f;
37 float fReciprical60 = 1.0f / 60.0f;
38 float fRed, fGreen, fBlue;
39 if (fHueAmount <= 60.0f) // Inc green
40 {
41 fRed = 1.0f;
42 fGreen = fHueAmount * fReciprical60;
43 fBlue = 0.0f;
44 }
45 else if (fHueAmount <= 120.0f) // Dec red
46 {
47 fRed = 1.0f - ((fHueAmount - 60.0f) * fReciprical60);
48 fGreen = 1.0f;
49 fBlue = 0.0f;
50 }
51 else if (fHueAmount <= 180.0f) // Inc blue
52 {
53 fRed = 0.0f;
54 fGreen = 1.0f;
55 fBlue = (fHueAmount - 120.0f) * fReciprical60;
56 }
57 else if (fHueAmount <= 240.0f) // Dec green
58 {
59 fRed = 0.0f;
60 fGreen = 1.0f - ((fHueAmount - 180.0f) * fReciprical60);
61 fBlue = 1.0f;
62 }
63 else if (fHueAmount <= 300.0f) // Inc red
64 {
65 fRed = (fHueAmount - 240.0f) * fReciprical60;
66 fGreen = 0.0f;
67 fBlue = 1.0f;
68 }
69 else // dec blue
70 {
71 fRed = 1.0f;
72 fGreen = 0.0f;
73 fBlue = 1.0f - ((fHueAmount - 300.0f) * fReciprical60);
74 }
75
76 // Now RGB is set, apply saturation
77 float fSaturationScaleR = 1.0f - fRed;
78 float fSaturationScaleG = 1.0f - fGreen;
79 float fSaturationScaleB = 1.0f - fBlue;
80 fRed += fSaturationScaleR * fSaturation;
81 fGreen += fSaturationScaleG * fSaturation;
82 fBlue += fSaturationScaleB * fSaturation;
83 clamp(fRed, 0.0f, 1.0f);
84 clamp(fGreen, 0.0f, 1.0f);
85 clamp(fBlue, 0.0f, 1.0f);
86
87 // Now saturation is added, apply brightness
88 float fBrightnessInv = 1.0f - fBrightness;
89 fRed -= fBrightnessInv;
90 fGreen -= fBrightnessInv;
91 fBlue -= fBrightnessInv;
92 clamp(fRed, 0.0f, 1.0f);
93 clamp(fGreen, 0.0f, 1.0f);
94 clamp(fBlue, 0.0f, 1.0f);
95
96 // Convert float to unsigned char
97 red = unsigned char(fRed * 255);
98 green = unsigned char(fGreen * 255);
99 blue = unsigned char(fBlue * 255);
100 }

102 void CColouruc::getHSB(float& fHue, float& fSaturation, float& fBrightness) const
103 {
104 // Convert unsigned char to float
105 float fOneOver255 = 1.0f / 255.0f;
106 float fRed = float(red) * fOneOver255;
107 float fGreen = float(green) * fOneOver255;
108 float fBlue = float(blue) * fOneOver255;
109
110 // Get maximum and minimum values of current RGB values
111 float cmax = std::max(fRed, std::max(fGreen, fBlue));
112 float cmin = std::min(fRed, std::min(fGreen, fBlue));
113 float diff = cmax - cmin; // diff of cmax and cmin.
114 fHue = -1, fSaturation = -1;
115
116 // If cmax and cmax are equal then fHue = 0
117 if (cmax == cmin)
118 fHue = 0;
119
120 // If cmax equal r then compute fHue
121 else if (cmax == fRed)
122 fHue = float(fmod(60 * ((fGreen - fBlue) / diff) + 360, 360));
123
124 // If cmax equal g then compute fHue
125 else if (cmax == green)
126 fHue = float(fmod(60 * ((fBlue - fRed) / diff) + 120, 360));
127
128 // If cmax equal b then compute fHue
129 else if (cmax == blue)
130 fHue = float(fmod(60 * ((fRed - fGreen) / diff) + 240, 360));
131
132 fHue /= 360.0f;
133
134 // If cmax equal zero
135 if (cmax == 0)
136 fSaturation = 0;
137 else
138 fSaturation = (diff / cmax);// *100;
139
140 // Compute fBrightness
141 fBrightness = cmax;// *100;
142 }
