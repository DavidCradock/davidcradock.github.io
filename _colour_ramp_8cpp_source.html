94 CColourf CColourRamp::getRampColour(float fRampPosition)
95 {
96 clamp(fRampPosition, 0.0f, 1.0f);
97
98 if (_mlistPoints.size() == 0)
99 {
100 return CColourf(1.0f, 1.0f, 1.0f, 1.0f); // Return white if no points
101 }
102
103 // Find the two closest points
104 unsigned int uiPointIndexLeft, uiPointIndexRight;
105 getAdjacentPointIndicies(fRampPosition, uiPointIndexLeft, uiPointIndexRight);
106
107 auto it = _mlistPoints.begin();
108 std::advance(it, uiPointIndexLeft);
109 const SPoint& pointLeft = *it;
110 std::advance(it, 1);
111 const SPoint& pointRight = *it;
112
113 // Linear interpolation between the two points
114 float t = (fRampPosition - pointLeft.fPosition) / (pointRight.fPosition - pointLeft.fPosition);
115 //return pointLeft.colour * (1.0f - t) + pointRight.colour * t;
116 return pointLeft.colour.interpolate(pointRight.colour, t);
117 }

119 void CColourRamp::getAdjacentPointIndicies(float fRampPosition, unsigned int& uiPointIndexLeft, unsigned int& uiPointIndexRight)
120 {
121 ThrowIfTrue(_mlistPoints.size() < 2, "Less than two points in the ramp");
122
123 // Find the insertion point
124 auto it = _mlistPoints.begin();
125 while (it != _mlistPoints.end() && it->fPosition < fRampPosition)
126 {
127 ++it;
128 }
129 uiPointIndexRight = (unsigned int)std::distance(_mlistPoints.begin(), it);
130 uiPointIndexLeft = uiPointIndexRight - 1;
131 if (uiPointIndexRight == 0)
132 {
133 uiPointIndexRight++;
134 uiPointIndexLeft = 0;
135 }
136 else if (uiPointIndexRight >= _mlistPoints.size())
137 {
138 uiPointIndexRight = (unsigned int)_mlistPoints.size() - 1;
139 uiPointIndexLeft = uiPointIndexRight - 1;
140 }
