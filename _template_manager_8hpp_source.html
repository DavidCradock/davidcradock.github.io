100 T* CManager<T>::add(const std::string& strObjectName)
101 {
102 // Find object
103 auto it = _mmapObjects.find(strObjectName);
104
105 // If object exists, increment reference count and return object
106 if (_mmapObjects.end() != it)
107 {
108 it->second->iRefCount++;
109 return it->second->pObject;
110 }
111
112 // Create new object and return it

114 ThrowIfFalse(pNewObject, "add(\"" + strObjectName + "\") failed. Unable to allocate memory for new object.");

125 // Find object

127
128 // If object doesn't exist, throw exception
129 ThrowIfTrue(_mmapObjects.end() == it, "remove(\"" + strObjectName + "\") failed. Object does not exist.");
130
131 // Decrement object's reference count
132 it->second->iRefCount--;
133
134 // If reference count is still high, simply return
135 if (it->second->iRefCount > 0)

138 // Remove object

140 delete it->second;
