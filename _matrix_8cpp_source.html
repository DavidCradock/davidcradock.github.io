346 void CMatrix::setProjectionPerspective(
347 float fWidth,
348 float fHeight,
349 float fFieldOfViewInDegrees,
350 float fNearClippingPlaneDistance,
351 float fFarClippingPlaneDistance,
352 float fAspectRatio)
353 {
354 float fFOVRadians = Core::deg2rad(fFieldOfViewInDegrees);
355 float fTanHalfFOV = tan(fFOVRadians * 0.5f);
356 float fNearMinusFar = fNearClippingPlaneDistance - fFarClippingPlaneDistance;
357 ThrowIfTrue(fNearMinusFar == 0.0f, "Given near and far clipping plane distances invalid.");
358
359 // Compute fAspectRatio if given a value <= 0.0f from the back buffer's framebuffer's resource dimensions
360 if (fAspectRatio <= 0.0f)
361 {
362 fAspectRatio = fWidth / fHeight;
363 }
364
365 // First column
366 m[0] = 1.0f / (fAspectRatio * fTanHalfFOV);
367 m[1] = 0.0f;
368 m[2] = 0.0f;
369 m[3] = 0.0f;
370
371 // Second column
372 m[4] = 0.0f;
373 m[5] = 1.0f / fTanHalfFOV;
374 m[6] = 0.0f;
375 m[7] = 0.0f;
376
377 // Third column
378 m[8] = 0.0f;
379 m[9] = 0.0f;
380 m[10] = fFarClippingPlaneDistance / (fNearMinusFar);
381 m[11] = -1.0f;
382
383 // Fourth column
384 m[12] = 0.0f;
385 m[13] = 0.0f;
386 m[14] = -(fFarClippingPlaneDistance * fNearClippingPlaneDistance) / (fFarClippingPlaneDistance - fNearClippingPlaneDistance);
387 m[15] = 1.0f; // 0.0f?
388 }

390 void CMatrix::setProjectionOrthographic(
391 float fWidth,
392 float fHeight,
393 float fLeft,
394 float fRight,
395 float fTop,
396 float fBottom,
397 float fNear,
398 float fFar)
399 {
400 // Compute fRight and fBottom if given a value <= 0.0f
401 if (fRight <= 0.0f || fBottom <= 0.0f)
402 {
403 fRight = fWidth;
404 fBottom = fHeight;
405 }
406
407 // First column
408 m[0] = 2.0f / (fRight - fLeft);
409 m[1] = 0.0f;
410 m[2] = 0.0f;
411 m[3] = 0.0f;
412
413 // Second column
414 m[4] = 0.0f;
415 m[5] = 2.0f / (fTop - fBottom);
416 m[6] = 0.0f;
417 m[7] = 0.0f;
418
419 // Third column
420 m[8] = 0.0f;
421 m[9] = 0.0f;
422 m[10] = -1.0f / (fFar - fNear);
423 m[11] = 0.0f;
424
425 // Fourth column
426 float fRightMinusLeft = fRight - fLeft;
427 if (fRightMinusLeft != 0)
428 m[12] = -(fRight + fLeft) / fRightMinusLeft;
429 else
430 Throw("A division by zero occurred.");
431 m[13] = -(fTop + fBottom) / (fTop - fBottom);
432 m[14] = -fNear / (fFar - fNear);
433 m[15] = 1.0f;
434 }

436 void CMatrix::setViewLookat(const CVector3f& vEyePos, const CVector3f& vTarget, const CVector3f& vUp)
437 {
438 // Forward vector
439 CVector3f vf = vTarget - vEyePos;
440 vf.normalise();
441
442 // Right (side) vector
443 CVector3f vs = vf.getCross(vUp);
444 vs.normalise();
445
446 // Up vector
447 CVector3f vu = vs.getCross(vf);
448 //vu.normalise(); // No need to normalise as other two vectors already are.
449
450 // First column
451 m[0] = vs.x;
452 m[1] = vu.x;
453 m[2] = -vf.x;
454 m[3] = 0.0f;
455
456 // Second column
457 m[4] = vs.y;
458 m[5] = vu.y;
459 m[6] = -vf.y;
460 m[7] = 0.0f;
461
462 // Third column
463 m[8] = vs.z;
464 m[9] = vu.z;
465 m[10] = -vf.z;
466 m[11] = 0.0f;
467
468 // Fourth column
