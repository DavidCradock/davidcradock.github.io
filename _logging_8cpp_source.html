120 void CLog::_mainThreadLoop(void)
121 {
122 bool bLogToSave = false;
123 CLogEntry logToSave;
124
125 while (!_mStop) // This gets set to true in the destructor
126 {
127 // Determine whether there's at least one log entry in the queue
128 // If so, copy it ready to save to file.
129 // We perform the minimal of stuff here to help prevent locking the mutex which could slow down calls to add()
130 bLogToSave = false;
131 _mQueueMutex.lock();
132 if (!_mQueueEntriesToAdd.empty())
133 {
134 bLogToSave = true;
135 logToSave = _mQueueEntriesToAdd.front();
136 _mQueueEntriesToAdd.pop();
137 }
138 _mQueueMutex.unlock();
139
140 // If there is a log entry, save it to file
141 if (bLogToSave)
142 {
143 std::ofstream file(_mstrFilename, std::ios::app);
144 if (file.is_open())
145 {
146 if (CLogEntry::ENTRY_TYPE_EMPTY_LINE == logToSave.eType)
147 {
148 file << _strTableColumnText[0] << ".";
149 file << _strTableColumnText[1];
150 file << _strTableColumnText[2];
151 file << _strTableColumnText[3];
152 file << _strTableColumnText[4];
153 file << "</div></td></tr>\n";
154 }
155 else if (CLogEntry::ENTRY_TYPE_HEADING == logToSave.eType)
156 {
157 file << _strTableColumnText[0];
158 file << _strTableColumnText[1];
159 file << _strTableColumnText[2] << "<br><b>" << logToSave.strText << "</b><br><br>";
160 file << _strTableColumnText[3];
161 file << _strTableColumnText[4];
162 file << "</div></td></tr>\n";
163 }
164 else if (CLogEntry::ENTRY_TYPE_NORMAL == logToSave.eType)
165 {
166 file << _strTableColumnText[0] << logToSave.strTime;
167 file << _strTableColumnText[1] << logToSave.strText;
168 file << _strTableColumnText[2] << logToSave.strFunctionName;
169 file << _strTableColumnText[3] << logToSave.strSourceLineNumber;
170 file << _strTableColumnText[4] << logToSave.strSourceFilename;
171 file << "</div></td></tr>\n";
172 }
173 }
174 else
175 {
176 }
177 }
178 else
179 {
180 // If there wasn't a log entry to save, let's sleep for a bit.
181 // Not sleeping due to there being a log entry makes it so that on the next
182 // loop is fast and allows us to prevent entries from queueing up.
183 std::this_thread::sleep_for(std::chrono::milliseconds(100));
184 }
185 }
186
187 // Now that we're stopping, write out any logs.
188 // Add one last entry...
189 LOG("Log file closing. Writing out the last of the added entries...");
190 while (!_mQueueEntriesToAdd.empty())
191 {
192 logToSave = _mQueueEntriesToAdd.front();
193 _mQueueEntriesToAdd.pop();
194
195 std::ofstream file(_mstrFilename, std::ios::app);
196 if (file.is_open())
197 {
198 file << _strTableColumnText[0] << logToSave.strTime;
199 file << _strTableColumnText[1] << logToSave.strText;
200 file << _strTableColumnText[2] << logToSave.strFunctionName;
201 file << _strTableColumnText[3] << logToSave.strSourceLineNumber;
202 file << _strTableColumnText[4] << logToSave.strSourceFilename;
203 file << "</div></td></tr>\n";
204 }
205 }
